\section{Les modèles de fondation, le cas de CrackSAM}
\frame{\sectionpage}

\begin{frame}{Le Mécanisme d'Attention des \emph{Transformers}\footnotemark[1]}
    L'attention calcule la contribution de chaque élément $j$ à la représentation de l'élément $i$ :
    \begin{equation*}
        \mathbf{y}_i = \sum_{j} a_{ij} \mathbf{v}_j \quad \text{avec} \quad a_{ij} = \frac{\exp\left(\frac{\mathbf{q}_i \, {}^t \mathbf{k}_j}{\sqrt{d_k}}\right)}{\sum_{l} \exp\left(\frac{\mathbf{q}_i \, {}^t \mathbf{k}_l}{\sqrt{d_k}}\right)}
    \end{equation*}
    \vskip 1ex
    \begin{block}{Composantes de l'attention}
        \begin{itemize}
            \item \textbf{\emph{Queries} ($\mathbf{q}_i$) :} Requête de l'élément $i$ (ce qu'il cherche)
            \item \textbf{\emph{Keys} ($\mathbf{k}_j$) :} Clé de l'élément $j$ (ce qu'il propose)
            \item \textbf{\emph{Values} ($\mathbf{v}_j$) :} Valeur de l'élément $j$ (son contenu sémantique)
            \item \textbf{Facteur $\sqrt{d_k}$ et poids $a_{ij}$ :} Facteur d'échelle et score d'attention.
        \end{itemize}
    \end{block}
    \footnotetext[1]{\nom{Vaswani} \etal, \og Attention Is All You Need\fg{}, \textit{NeurIPS}, 2017.}
\end{frame}

\begin{frame}{\emph{Vision Transformers} (\emph{ViT}) : De l'Image aux Patchs\footnotemark[2]}
    L'architecture \emph{ViT} adapte les \emph{Transformers} au traitement d'images en découpant l'image en une séquence de morceaux :
    \vskip 1ex
    \begin{block}{Étapes clés de traitement}
        \begin{enumerate}
            \item \textbf{Découpage en patchs} : L'image est divisée en une grille de patchs de $16 \times 16$ pixels (qui jouent le rôle de mots).
            \item \textbf{Projection linéaire (\emph{Patch Embedding})} : Chaque patch est projeté vectoriellement dans un espace sémantique de dimension fixe.
            \item \textbf{\emph{Embeddings} de Position} : Ajout d'une coordonnée de position à chaque patch pour conserver la géométrie 2D de l'image.
            \item \textbf{Jeton sémantique global} : Ajout d'un jeton virtuel en début de séquence pour résumer toute l'image.
        \end{enumerate}
    \end{block}
    \footnotetext[2]{\nom{Dosovitskiy} \etal, \og An Image is Worth 16x16 Words: Transformers for Image Recognition at Scale\fg{}, \textit{ICLR}, 2021.}
\end{frame}

\begin{frame}{Modèles de Fondation \& \emph{Transformers}}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Trois piliers des modèles de fondation}
                Un modèle unique pré-entraîné à grande échelle pour être adapté à de nombreuses tâches :
                \begin{itemize}
                    \item \textbf{\emph{Scale}} : taille du modèle (milliards de paramètres) et des données.
                    \item \textbf{Auto-supervision} : pré-entraînement autonome sans annotations.
                    \item \textbf{Adaptabilité (\emph{Fine tuning})} : socle commun réutilisable.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{Rôle des \emph{Transformers}}
                L'architecture \emph{Transformer} (\emph{ViT}, \emph{GPT}) domine largement ce domaine.
                \vskip 1ex
                \textbf{Avantages :}
                \begin{itemize}
                    \item Parallélisation massive (entraînement rapide sur \emph{GPU}).
                    \item Capacité de généralisation exceptionnelle.
                \end{itemize}
                \textbf{Limites :}
                \begin{itemize}
                    \item Complexité quadratique $O(N^2)$ sur la longueur des séquences.
                    \item Modèles complexes agissant comme des \og boîtes noires \fg{}.
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{SAM : \emph{Segment Anything Model}\footnotemark[3]}
    \begin{columns}
        \begin{column}{0.56\textwidth}
            \begin{block}{Architecture asymétrique}
                \begin{itemize}
                    \item \textbf{\emph{Image Encoder}} : Un grand \emph{ViT} (ViT-H, 637M paramètres) pré-entraîné (\emph{MAE}). Extrait un \emph{embedding} d'image de taille $64 \times 64$.
                    \item \textbf{\emph{Prompt Encoder}} : Module très léger. Encode les invites spatiales (points, boîtes via codage positionnel) ou textuelles (via \emph{CLIP}).
                    \item \textbf{\emph{Mask Decoder}} : Décodeur rapide par \emph{Transformer} bidirectionnel. Génère les masques en temps réel ($\sim 50$ ms).
                    \item \textbf{\emph{Dataset} SA-1B} : 11M d'images, 1,1 milliard de masques de haute qualité.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.41\textwidth}
            \centering

            \vskip 1ex
            {\scriptsize Schéma de SAM : Séparation stricte entre le calcul lourd (image) et le calcul rapide (invites).}
        \end{column}
    \end{columns}
    \footnotetext[3]{\nom{Kirillov} \etal, \og Segment Anything\fg{}, \textit{ICCV}, 2023. Code : \texttt{github.com/facebookresearch/segment-anything}}
\end{frame}

\begin{frame}{CrackSAM : \emph{Fine-Tuning} Efficace par \emph{LoRA}}
    \begin{block}{Difficultés de la détection de fissures}
        \begin{itemize}
            \item \textbf{Déséquilibre extrême} : Les fissures fines ne représentent qu'une fraction infime des pixels de l'image.
            \item \textbf{Perturbations complexes} : Présence d'ombres, de flous, d'humidité et textures rugueuses du béton.
            \item \textbf{Changement de domaine} : Les scènes de génie civil diffèrent fortement du pré-entraînement de SAM.
        \end{itemize}
    \end{block}
    \vskip 1ex
    \begin{block}{Principe de la \emph{Low-Rank Adaptation} (\emph{LoRA})}
        On part de la matrice de poids pré-entraînée $W_0$ (gelée) et on cherche deux matrices $A$ et $B$ de petit rang $r$ telles que :
        \begin{equation*}
            W = W_0 + \Delta W \quad \text{avec} \quad \Delta W = B \cdot A
        \end{equation*}
        \textbf{Avantages :}
        \begin{itemize}
            \item Seules les petites matrices $A$ et $B$ sont entraînées, tandis que $W_0$ reste figée.
            \item Division par plus de 100 du nombre de paramètres à entraîner ($\sim 4,4$M au lieu de $637$M).
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Introduction et Problématique}
    \begin{block}{Auscultation des structures et risques géologiques}
        \begin{itemize}
            \item L'évaluation de l'intégrité des infrastructures (ponts, routes, centrales nucléaires) et la surveillance des risques géologiques (failles, glissements de terrain) requièrent une cartographie précise des fissures.
            \item Le relevé manuel est chronophage, subjectif et difficile à généraliser à grande échelle.
        \end{itemize}
    \end{block}
    \begin{block}{Limites des approches par Apprentissage Profond (\emph{Deep Learning})}
        \begin{itemize}
            \item Les modèles de segmentation actuels (\eg CrackSegDiff) obtiennent de très bons scores mais se comportent comme des \og boîtes noires \fg{}.
            \item Dépendance critique à de grands volumes de données annotées par des experts.
            \item \textbf{Manque de robustesse} face aux décalages de domaine (nouveaux capteurs, textures inédites, bruits inconnus).
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Objectifs et Contributions}
    \begin{block}{Notre approche : universelle, géométrique et sans apprentissage}
        Nous proposons une méthode non supervisée qui fusionne des données multi-modales (optique et profondeur) dans une représentation de graphe géométrique robuste.
    \end{block}
    \begin{block}{Contributions principales}
        \begin{enumerate}
            \item \textbf{Fusion au niveau de l'opérateur (Hessien) :} Combinaison des matrices Hessiennes normalisées pour exploiter la cohérence géométrique (vecteurs propres) des fissures à travers les modalités.
            \item \textbf{Graphe de Frangi Généralisé :} Modélisation sous forme de graphe spatial à grand rayon ($R=10$ px) codant la similarité géométrique locale (élongation, contraste, alignement).
            \item \textbf{Algorithme d'extraction topologique :} Prétraitement par double seuillage, réduction par arbre couvrant minimal (MST) et squelettisation par centralité de degré pondérée.
            \item \textbf{Robustesse et validation :} Stabilité face aux bruits intenses sur le benchmark FIND et transférabilité en zéro-shot (Palais des Papes).
        \end{enumerate}
    \end{block}
\end{frame}

\begin{frame}{Fusion d'Hessiennes Multi-Échelles}
    \begin{block}{Principe de la fusion géométrique}
        \begin{itemize}
            \item Soient les modalités d'entrée $\mathcal{I} = \{I^{(\text{Int})}, I^{(\text{Rng})}, \ldots\}$ (Intensité, Range, etc.).
            \item Pour chaque modalité $m$ et échelle $\sigma \in \Sigma$, on calcule la matrice Hessienne $\mathcal{H}_\sigma^{(m)}(\mathbf{x})$ (convolutions gaussiennes).
            \item Normalisation pour s'affranchir de la dynamique des capteurs :
            \begin{equation*}
                \hat{\mathcal{H}}_{\sigma}^{(m)}(\mathbf{x}) = \frac{\mathcal{H}_{\sigma}^{(m)}(\mathbf{x})}{\max_{\mathbf{y}} \|\mathcal{H}_{\sigma}^{(m)}(\mathbf{y})\|}
            \end{equation*}
            \item Le Hessien fusionné est une combinaison linéaire des matrices normalisées :
            \begin{equation*}
                \mathcal{H}_{\sigma}^{\text{fused}}(\mathbf{x}) = \sum_{m} w_m \hat{\mathcal{H}}_{\sigma}^{(m)}(\mathbf{x})
            \end{equation*}
            \item Décomposition spectrale : valeurs propres $\lambda_1, \lambda_2$ ($|\lambda_1| \le |\lambda_2|$) et vecteurs propres associés.
            \item Prior physique : les fissures sont modélisées comme des \og vallées sombres \fg{} ($\lambda_2 > 0$).
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Graphe de Similarité de Frangi Généralisé}
    On définit un graphe spatial $G=(V, E)$ où les n\oe{}uds $V$ sont les pixels et les arêtes $E$ connectent les voisins dans un rayon $R = 10\text{ px}$ (permettant de franchir les discontinuities dues au bruit).
    \begin{block}{Similarité d'arête $S_{ij} = \max_{\sigma} ( \mathcal{S}_{\text{shape}}^\sigma \cdot \mathcal{S}_{\text{int}}^\sigma \cdot \mathcal{S}_{\text{align}}^\sigma ) \in [0, 1]$}
        \begin{itemize}
            \item \textbf{Élongation ($\mathcal{S}_{\text{shape}}$) :} Utilise le ratio de Frangi $\mathcal{R}_B = |\lambda_1|/|\lambda_2|$ pour rejeter les structures de type bulles/blobs.
            \item \textbf{Intensité/Contraste ($\mathcal{S}_{\text{int}}$) :} Favorise les zones à forte courbure ($\lambda_2$) de manière multiplicative (les deux n\oe{}uds doivent être sur la fissure).
            \item \textbf{Alignement géométrique ($\mathcal{S}_{\text{align}}$) :} Colle topologique pénalisant la désorientation des vecteurs propres principaux ($\delta_\theta$).
        \end{itemize}
    \end{block}
    \begin{block}{Dissimilarité pondérée par la distance euclidienne}
        \begin{equation*}
            d_{ij} = (1 - S_{ij}) \|\mathbf{x}_i - \mathbf{x}_j\|
        \end{equation*}
        Agit comme un régulariseur empêchant l'arbre couvrant de faire des sauts erratiques.
    \end{block}
\end{frame}

\begin{frame}{Extraction Topologique et Squelettisation}
    L'extraction se fait en 4 étapes clés :
    \begin{block}{Algorithme d'extraction}
        \begin{enumerate}
            \item \textbf{Double Seuillage :} Filtrage des n\oe{}uds sur la courbure $\lambda_2$ puis des arêtes sur la similarité $S_{ij}$ pour éliminer le bruit.
            \item \textbf{Arbre Couvrant Minimal (MST) :} Réduction du graphe sur la composante principale. Le MST simplifie le réseau en une structure arborescente légère tout en conservant les connexions les plus fortes.
            \item \textbf{Centralité de Degré Pondérée (Weighted Betweenness) :} Calcul rapide sur l'arbre. Les n\oe{}uds situés sur l'axe central de la fissure accumulent un grand nombre de chemins minimaux et obtiennent un score de centralité élevé.
            \item \textbf{Élagage Adaptatif :} Suppression des branches secondaires à faible centralité pour obtenir un squelette précis de 1 pixel d'épaisseur.
        \end{enumerate}
    \end{block}
\end{frame}

\section{Que peut-on faire ?}

\begin{frame}{Protocole Expérimental et Données}
    \begin{block}{Le Benchmark FIND}
        \begin{itemize}
            \item 500 images de taille $256 \times 256\text{ px}$.
            \item Propose deux modalités parfaitement enregistrées : Intensité (optique) et Range (profondeur).
            \item Nous excluons 23 images dont les annotations de la vérité terrain sont inconsistantes avec la géométrie physique des fissures.
        \end{itemize}
    \end{block}
    \begin{block}{Métriques d'Évaluation}
        \begin{itemize}
            \item \textbf{Indice de Jaccard (IoU)} : Mesure globale de recouvrement.
            \item \textbf{Indice de Tversky} ($\alpha = 1, \beta = 0.5$) : Pénalise moins les faux positifs que les faux négatifs (priorité à la détection de toutes les fissures pour des raisons de sécurité).
            \item \textbf{Distance de Wasserstein} : Mesure du travail de transport physique entre la prédiction et la vérité terrain, robuste aux petits décalages spatiaux.
            \item Pour les indices de Jaccard/Tversky sur des squelettes d'un pixel, une dilatation morphologique de 3 pixels est appliquée.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Conclusion et Perspectives}
    \begin{block}{Synthèse des travaux}
        \begin{itemize}
            \item \textbf{Sans apprentissage et universel :} Une alternative robuste aux approches Deep Learning boîte-noire.
            \item \textbf{Fusion géométrique précoce :} Intégration naturelle de modalités physiques complémentaires par l'opérateur Hessien.
            \item \textbf{Robustesse extrême au bruit :} Les relations de voisinage et l'alignement géométrique préservent la continuité topologique là où les réseaux de neurones échouent.
            \item \textbf{Outil d'annotation :} Permet de générer des squelettes précis modifiables pour assister l'annotation humaine et l'entraînement de futurs modèles.
        \end{itemize}
    \end{block}
    \begin{block}{Perspectives de recherche}
        \begin{itemize}
            \item Validation sur la future grande base de données \textbf{3DCrack}.
            \item Importance capitale du prétraitement (comme la modalité "Filtered") pour maximiser la fidélité topologique.
            \item Extension du paradigme de graphe informationnel à d'autres structures tubulaires (médicales, réseaux de transport).
        \end{itemize}
    \end{block}
\end{frame}
